**Problem Statement:** Given a string `a` of length `n` and a string `b` of length `len`, count how many substrings of `a` with length `len` satisfy: after arbitrary matching with `b`, each pair of values sums to at least `h`.

## Solution

**Key Insight:** Sort `b` in descending order. For each element in `a`, find the largest index `i` such that `b[i] + a[j] ≥ h`. This means `b[1]` to `b[i]` can all pair with `a[j]`.

**Algorithm:**

1. For each position in `a`, compute how many elements from `b` it can match with
2. For each sliding window of length `len` in `a`, we need to check if a valid matching exists
3. Use a segment tree to maintain the matching constraints efficiently

**Segment Tree Approach:**

- Initially, subtract `i` from each position `i` (0-indexed to len-1)
- For each element in the current window, add 1 to range `[0, c[i]]` where `c[i]` is its match count
- If the maximum value in the segment tree ≤ 0, the window is valid
- Slide the window: remove the leftmost element and add the new right element

**Correctness:** If there exists any position where `c[i] > i`, we cannot find a valid matching because we need to match `i+1` elements from `b[0..i]`, which is impossible.

```cpp
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<queue>
#include<algorithm>
#include<vector>
#include<iostream>
#define max(x,y)(x>y?x:y)
#define min(x,y)(x<y?x:y)
#define ll long long
#define clr(x,y) memset(x,y,sizeof(x))
using namespace std;

const int inf = 1e9;
const int N = 200000;
int a[N],b[N];
int n,m,h;
int ma[4*N],add[4*N];

bool cmp(int a,int b){
    return a>b;
}

void pushdown(int x){
    add[x+x] += add[x];
    add[x+x+1] += add[x];
    add[x] = 0;
}
void pushup(int x){
    ma[x] = max(ma[x+x]+add[x+x],ma[x+x+1]+add[x+x+1]);
}

void insert(int x,int l,int r,int tl,int tr,int v){
    if(l>tr || r<tl) return ;
    if(l<=tl && r>=tr){
        add[x] += v;
        return ;
    }
    pushdown(x);
    int mid = (tr+tl)/2;
    insert(x*2,l,r,tl,mid,v);
    insert(x*2+1,l,r,mid+1,tr,v);
    pushup(x);
}

int main(){
    scanf("%d%d%d",&n,&m,&h);
    for(int i=1;i<=m;i++) scanf("%d",b+i);
    sort(b+1,b+m+1,cmp);
    
    // For each element in a, find how many b elements it can match with
    for(int i=1;i<=n;i++){
        int tmp;
        scanf("%d",&tmp);
        int l = 1, r = m+1;
        while(r!=l){
            int mid = (l+r)>>1;
            if(b[mid] + tmp >=h) l = mid+1;
            else r = mid;
        }
        l--;
        a[i] = l;
    }
    
    // Initialize segment tree
    for(int i=0;i<=m;i++) insert(1,i,i,0,m,-i);
    for(int i=1;i<=m;i++) insert(1,a[i],m,0,m,1);
    
    int ans = 0;
    for(int i=m;i<=n;i++){
        if((ma[1] + add[1])<=0) ans++;
        if(i!=n){
            insert(1,a[i-m+1],m,0,m,-1);
            insert(1,a[i+1],m,0,m,1);
        }
    }
    printf("%d\n",ans);
    return 0;
}
```

**Time Complexity:** O(n log m)
**Space Complexity:** O(n)
